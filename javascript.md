### Control flow statements:
```javascript
if (x > 3 || x < 3) {
    console.log("The value of 'x' is not 3!");
}

if (x > 2 && x < 4) {
    console.log("The value of 'x' is 3!");
}

if (color === 'green') {
    console.log('Color is green!');
} else if (color === 'yellow') {
    console.log('Color is yellow!');
} else {
    console.log('Color was not detected!');
}

for (let i = 0; i <= 10; i++) {
    console.log(i);
}

for (let todo of todos) {
    console.log(todo)
}

let x = 0;
while (x <= 10) {
    console.log(`The value of 'x' is ${x}.`);
    x++;
}

switch (x) {
    case 1:
        console.log('The value of "x" is 1!');
        break;
    case 2:
        console.log('The value of "x" is 2!');
        break;
    default:
        console.log('The value of "x" was not detected!');
        break;
}
```

### Functions
```javascript
function sum(x, y) {
    return x + y;
}

const sum = (x, y) => {
    return x + y;
}
```
